void sort(int a[], int size) {
    int i;
    int j;
    int tmp;
    
    for (i = 0; i < size-1; i += 1) {
        for (j = i; j < size; j += 1) {
            if (a[i] > a[j]) {
                tmp = a[i];
                a[i] = a[j];
                a[j] = tmp;
            }
        }
    }
}

int main(void) {
    
    int i;
    int j;
    int a[] = {5, 2, 7, 9, 0, 4, 6, 1, 3, 8};
    
    sort(a, 10);
    
    for (i = 0; i < 10; i += 1) {
        j = a[i];
    }
  
}
